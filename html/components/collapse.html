    $('.collapse__button').on('click', function(event){
      var $target = $(this).siblings('.collapse__content');

      if ($target.is(':hidden')) {
        $target.slideDown(100).parent().siblings().find('.collapse__content').slideUp(100);
      }
      else {
        $target.slideUp(100);
      }
    });
