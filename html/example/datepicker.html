                  ;(function(){
                    var calender = function(){
                      var callback = ['select', 'update'];

                      function getData(context){
                        var array = []
                          , date;

                        do {
                          date = new Date([context.year, context.month, array.length+1].join());
                          array.push({
                            year: context.year,
                            month: context.month,
                            date: date.getDate(),
                            week: date.getDay()
                          });
                        }
                        while(context.month == date.getMonth()+1);

                        array.pop();

                        while(array[0].week != 0){
                          array.unshift({
                            year: context.year,
                            month: context.month,
                            date: null,
                            week: array[0].week-1
                          });
                        }

                        while(array[array.length-1].week != 6){
                          array.push({
                            year: context.year,
                            month: context.month,
                            date: null,
                            week: array[array.length-1].week+1
                          });
                        }

                        return array;
                      };

                      function getHead(){
                        var weeks = ['일', '월', '화', '수', '목', '금', '토'];
                        var thead = document.createElement('thead');
                        var row = document.createElement('tr');
                        var col;

                        while(weeks.length){
                          col = document.createElement('th');
                          col.setAttribute('class', 'calender-week');
                          col.setAttribute('scope', 'col');
                          col.innerText = weeks[0];
                          row.append(col);
                          weeks.shift();
                        }

                        thead.append(row)

                        return thead;
                      }

                      function getBody(context){
                        var data = context.data.slice();
                        var tbody = document.createElement('tbody');
                        var row, col, button;

                        while(data.length){
                          if (data[0].week == 0) {
                            row = document.createElement('tr');
                            tbody.append(row);
                          }

                          col = document.createElement('td');
                          col.setAttribute('class', 'calender-date');
                          row.append(col);

                          if (data[0].date) {
                            button = document.createElement('button');
                            button.setAttribute('class', 'calender-button');
                            button.dataset.date = [data[0].year, data[0].month, data[0].date].join('-');
                            button.dataset.week = data[0].week;
                            button.innerText = data[0].date;
                            button.addEventListener('click', function(event){
                              context.callback.select.call(context, event);
                            });

                            col.append(button);
                          }

                          data.shift();
                        }

                        return tbody;
                      }

                      function getTable(context){
                        var table = document.createElement('table');
                        var caption = document.createElement('caption');

                        caption.setAttribute('class', 'calender-caption');
                        caption.innerText = [context.year, context.month].join('.');

                        table.setAttribute('class', 'calender-table');
                        table.append(caption);
                        table.append(getHead());
                        table.append(getBody(context));

                        return table;
                      }

                      function getCtlr(context){
                        var result = document.createElement('div');
                        var button;
                        var array = [
                          { text: '<<', func: function(event){ update(context, 'year', -1) } },
                          { text: '<', func: function(event){ update(context, 'month', -1) } },
                          { text: '>', func: function(event){ update(context, 'month', 1) } },
                          { text: '>>', func: function(event){ update(context, 'year', 1) } }
                        ];

                        result.setAttribute('class', 'calender-ctlr');

                        while(array.length){
                          button = document.createElement('button');
                          button.innerText = array[0].text;
                          button.addEventListener('click', array[0].func);
                          result.append(button);
                          array.shift();
                        }

                        return result;
                      }

                      function update(context, string, number){
                        context[string] += number;

                        if (context.month >= 13) {
                          context.year++;
                          context.month = 1;
                        }

                        if (context.month <= 0) {
                          context.year--;
                          context.month = 12;
                        }

                        context.data = getData(context);
                        context.table.querySelector('caption').innerText = [context.year, context.month].join('.');
                        context.table.querySelector('tbody').replaceWith(getBody(context));
                        context.callback.update.call(context, context.table);
                      }

                      function Calender(target){
                        this.target = target;
                        this.year = new Date().getFullYear();
                        this.month = new Date().getMonth()+1;
                        this.data = getData(this);
                        this.table = getTable(this);
                        this.button;

                        this.callback = function(array){
                          var result = {};

                          while (array.length) {
                            result[array.shift()] = function(){};
                          }

                          return result;
                        }(callback.slice());

                        this.target.append(this.table);
                        this.target.append(getCtlr(this));
                      }

                      Calender.prototype = {
                        on: function(property, fn){
                          this.callback[property] = fn;
                        }
                      };

                      return {
                        create: function(target){
                          return new Calender(target);
                        }
                      }
                    }();

                    var calender = calender.create(document.querySelector('.calender'));

                    calender.on('select', function(event){
                      console.log(event.target, this);
                    });

                    calender.on('update', function(table){
                      console.log(table, this);
                    });
                  }());
